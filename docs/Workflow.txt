------
# "New Computer" installation Checklist workflow:

# Clone the repo (Use the SSH URL now)
git clone git@github.com:tl-uk/CTT.git
cd CTT

# 1. Reconstruct the C++ Environment (Uses your Brewfile)
brew bundle

# 2. Reconstruct the Python Environment (Uses your requirements.txt)
uv venv --python 3.13
source .venv/bin/activate
uv pip install -r requirements.txt

# 3. Build the Engine
cmake -B build -S .
cmake --build build

------
# Run a Clean Build
rm -rf build
cmake -B build -S . -DCMAKE_OSX_ARCHITECTURES="arm64" -DCMAKE_POLICY_VERSION_MINIMUM=3.5
cmake --build build
# Check if the final binary was linked successfully:
ls -l build/CTT_Engine
# Check arm64
file build/CTT_Engine
# Run the executable
./build/CTT_Engine

# Clean start
rm -rf .venv
# Use 3.13 (Native arm64 and highly stable)
uv venv --python 3.13
source .venv/bin/activate
# Download pre-built files
uv pip install bptk-py pyzmq eclipse-sumo
